 let Tf:=fresh in
 assert (Tf:∃ K, (BetS G M K ∧ BetS D H K)) by (conclude postulate_Euclid5);destruct Tf as [K];spliter.
 assert (Col G A M) by (conclude lemma_rayimpliescollinear).
 assert (Col G M K) by (conclude_def Col ).
